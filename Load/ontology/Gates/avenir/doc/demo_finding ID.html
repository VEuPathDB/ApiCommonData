1. Data loading

#Load datasets
datanames<-list.files(here::here("rawdata"))
dflist<-list()

  dflist[[i]]<-rio::import(here::here("rawdata",i))
}
x<-names(dflist)

2. Finding the potential ID/IDs

dfid<-list()
dfid <- map(dflist, ~ .x %>% 
        mutate(across(where(is.numeric), as.character))%>%
        skim()%>%
        as_tibble()%>%
        filter(character.n_unique >= n()*0.7))

namelist<-unique(unlist(sapply(dfid, "[[", "skim_variable")))

for (i in 1:4){
    dflist[[i]]%>%
      mutate_if(is.numeric,as.character)%>%
      skim()%>%
      as_tibble()%>%
      filter(skim_variable %in% namelist2)%>%
      select(variable_name=skim_variable, n_missing, complete_rate, n_uniqueID=character.n_unique)%>%
      mutate(N_total=nrow(dflist[[i]]))%>%
      kbl(caption = names(dflist[i]),digits=2,desc=T)%>%
         kable_classic(full_width=F, html_font = "Cambria")%>%
      print(kbl())
}
